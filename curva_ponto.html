    void CirclePoints(int x, int y, int color){
        write_pixel( x, y, color);
        write_pixel( x, -y, color);
        write_pixel(-x, y, color);
        write_pixel(-x, -y, color);
        write_pixel( y, x, color);
        write_pixel( y, -x, color);
        write_pixel(-y, x, color);
        write_pixel(-y, -x, color);
    }/* end CirclePoints */

    void MidPointCircle(int r, int color){
        int x, int y;
        float d;
        /* Valores iniciais */
        x = 0;
        y = r;
        d = 5/4 - r;
        CirclePoints(x, y, color);
        while (x > y){
            if (d < 0){
                /* Selecione E */
                d = d + 2 * x + 3;
                x++;
            }else{
                /* Selecione SE */
                d = d + 2 * (x - y) + 5;
                x++;
                y--;
            }/*end if*/
            CirclePoints(x, y, color);
        }/* end while */
    }/* end MidpointCircle */
